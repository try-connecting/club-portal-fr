    let currentQuiz = null;
    let currentQuestionIndex = 0;
    let currentPartner = 'A';
    let answers = { A: {}, B: {} };
    let selectedAnswer = null;

    function startQuiz(quizType) {
      currentQuiz = quizData[quizType];
      currentQuestionIndex = 0;
      currentPartner = 'A';
      answers = { A: {}, B: {} };
      selectedAnswer = null;

      // Hide quiz selection
      document.getElementById('quizSelection').style.display = 'none';
      document.getElementById('quizInterface').style.display = 'block';
      document.getElementById('resultsScreen').style.display = 'none';

      // Set quiz title
      document.getElementById('quizTitle').textContent = currentQuiz.title;

      // Show/hide partner selection
      const partnerSelection = document.getElementById('partnerSelection');
      if (currentQuiz.dualPartner) {
        partnerSelection.classList.remove('hidden');
      } else {
        partnerSelection.classList.add('hidden');
      }

      displayQuestion();
    }

    function displayQuestion() {
      // For dual-partner quizzes, calculate actual question index based on partner
      let actualQuestionIndex = currentQuestionIndex;
      let totalQuestionsForPartner = currentQuiz.questions.length;
      
      if (currentQuiz.title.includes('Love Language')) {
        totalQuestionsForPartner = 10; // Each partner gets 10 questions
        if (currentPartner === 'B') {
          actualQuestionIndex = currentQuestionIndex + 10; // Partner B gets questions 10-19
        }
        // Partner A gets questions 0-9 (currentQuestionIndex stays the same)
      } else if (currentQuiz.title.includes('Conflict Style')) {
        totalQuestionsForPartner = 15; // Each partner gets 15 questions
        if (currentPartner === 'B') {
          actualQuestionIndex = currentQuestionIndex + 15; // Partner B gets questions 15-29
        }
        // Partner A gets questions 0-14 (currentQuestionIndex stays the same)
      }
      
      const question = currentQuiz.questions[actualQuestionIndex];
      document.getElementById('questionText').textContent = question.question;

      // Update progress
      const progress = ((currentQuestionIndex + 1) / totalQuestionsForPartner) * 100;
      document.getElementById('progressBar').style.width = progress + '%';
      document.getElementById('progressText').textContent = `Question ${currentQuestionIndex + 1} of ${totalQuestionsForPartner}`;

      // Clear and populate answers
      const answersContainer = document.getElementById('answersContainer');
      answersContainer.innerHTML = '';

      if (currentQuiz.title.includes('Health Score')) {
        // Scale answers for health score quiz
        for (let i = 1; i <= 5; i++) {
          const answerDiv = document.createElement('div');
          answerDiv.className = 'answer-option';
          answerDiv.innerHTML = `
            <span>${getScaleText(i)}</span>
            <div class="scale-number">${i}</div>
          `;
          answerDiv.onclick = (e) => selectAnswer(e.currentTarget, i);
          answersContainer.appendChild(answerDiv);
        }
      } else {
        // Multiple choice answers
        question.answers.forEach((answer, index) => {
          const answerDiv = document.createElement('div');
          answerDiv.className = 'answer-option';
          answerDiv.textContent = answer.text;
          answerDiv.onclick = (e) => selectAnswer(e.currentTarget, index);
          answersContainer.appendChild(answerDiv);
        });
      }

      // Update navigation buttons
      document.getElementById('backBtn').style.display = currentQuestionIndex > 0 ? 'inline-flex' : 'none';
      document.getElementById('nextBtn').disabled = true;
    }

    function getScaleText(value) {
      const scaleTexts = {
        1: 'Strongly Disagree',
        2: 'Disagree',
        3: 'Neutral',
        4: 'Agree',
        5: 'Strongly Agree'
      };
      return scaleTexts[value];
    }

    function selectAnswer(clickedElement, answerIndex) {
      // Remove previous selection
      document.querySelectorAll('.answer-option').forEach(answer => {
        answer.classList.remove('selected');
      });

      // Mark new selection using the clicked element directly
      clickedElement.classList.add('selected');
      selectedAnswer = answerIndex;

      // Enable next button
      document.getElementById('nextBtn').disabled = false;
    }

    function nextQuestion() {
      if (selectedAnswer === null) return;

      // Store answer with actual question index
      let actualQuestionIndex = currentQuestionIndex;
      if (currentQuiz.title.includes('Love Language') && currentPartner === 'B') {
        actualQuestionIndex = currentQuestionIndex + 10; // Partner B answers questions 10-19
      } else if (currentQuiz.title.includes('Conflict Style') && currentPartner === 'B') {
        actualQuestionIndex = currentQuestionIndex + 15; // Partner B answers questions 15-29
      }

      if (currentQuiz.dualPartner) {
        if (!answers[currentPartner][actualQuestionIndex]) {
          answers[currentPartner][actualQuestionIndex] = selectedAnswer;
        }
      } else {
        answers.A[currentQuestionIndex] = selectedAnswer;
      }

      selectedAnswer = null;

      // Determine max questions for current partner
      let maxQuestionsForPartner = currentQuiz.questions.length;
      if (currentQuiz.title.includes('Love Language')) {
        maxQuestionsForPartner = 10; // Each partner gets 10 questions
      } else if (currentQuiz.title.includes('Conflict Style')) {
        maxQuestionsForPartner = 15; // Each partner gets 15 questions
      }

      if (currentQuestionIndex < maxQuestionsForPartner - 1) {
        currentQuestionIndex++;
        displayQuestion();
      } else {
        // Quiz complete for this partner
        if (currentQuiz.dualPartner && currentPartner === 'A') {
          // Switch to partner B
          currentPartner = 'B';
          currentQuestionIndex = 0;
          document.querySelector('[data-partner="B"]').classList.add('active');
          document.querySelector('[data-partner="A"]').classList.remove('active');
          displayQuestion();
        } else {
          // Show results
          showResults();
        }
      }
    }

    function previousQuestion() {
      if (currentQuestionIndex > 0) {
        currentQuestionIndex--;
        displayQuestion();
      }
    }

    function showResults() {
      document.getElementById('quizInterface').style.display = 'none';
      document.getElementById('resultsScreen').style.display = 'block';
      document.getElementById('resultsTitle').textContent = 'Your Results';

      const resultsContent = document.getElementById('resultsContent');
      resultsContent.innerHTML = generateResults();
    }

    function generateResults() {
      if (currentQuiz.title.includes('Love Language')) {
        return generateLoveLanguageResults();
      } else if (currentQuiz.title.includes('Health Score')) {
        return generateHealthScoreResults();
      } else if (currentQuiz.title.includes('Conflict Style')) {
        return generateConflictStyleResults();
      }
    }

    function generateLoveLanguageResults() {
      const partnerA = { words: 0, service: 0, gifts: 0, time: 0, touch: 0 };
      const partnerB = { words: 0, service: 0, gifts: 0, time: 0, touch: 0 };

      // Count answers for Partner A (questions 0-9)
      Object.entries(answers.A).forEach(([questionIndex, answerIndex]) => {
        const question = currentQuiz.questions[parseInt(questionIndex)];
        const answer = question.answers[answerIndex];
        partnerA[answer.type]++;
      });

      // Count answers for Partner B (questions 10-19)  
      Object.entries(answers.B).forEach(([questionIndex, answerIndex]) => {
        const question = currentQuiz.questions[parseInt(questionIndex)];
        const answer = question.answers[answerIndex];
        partnerB[answer.type]++;
      });

      // Get primary love languages
      const primaryA = Object.keys(partnerA).reduce((a, b) => partnerA[a] > partnerA[b] ? a : b);
      const primaryB = Object.keys(partnerB).reduce((a, b) => partnerB[a] > partnerB[b] ? a : b);

      const languageNames = {
        words: 'Words of Affirmation',
        service: 'Acts of Service',
        gifts: 'Receiving Gifts',
        time: 'Quality Time',
        touch: 'Physical Touch'
      };

      return `
        <div class="partner-results">
          <div class="love-language-result">
            <h3>Partner A</h3>
            <div class="score">${languageNames[primaryA]}</div>
            <p class="description">Your primary love language is ${languageNames[primaryA].toLowerCase()}. You feel most loved when your partner expresses affection this way.</p>
          </div>
          <div class="love-language-result">
            <h3>Partner B</h3>
            <div class="score">${languageNames[primaryB]}</div>
            <p class="description">Your primary love language is ${languageNames[primaryB].toLowerCase()}. You feel most loved when your partner expresses affection this way.</p>
          </div>
        </div>
        <div class="love-language-result">
          <h3>Understanding Each Other</h3>
          <p class="description">Now that you know each other's love languages, focus on expressing love in the way your partner receives it best. This understanding can transform your relationship!</p>
        </div>
      `;
    }

    function generateHealthScoreResults() {
      const categories = currentQuiz.categories;
      const scores = {};
      let totalScore = 0;

      categories.forEach(category => {
        scores[category] = 0;
      });

      Object.values(answers.A).forEach((score, index) => {
        const question = currentQuiz.questions[index];
        scores[question.category] += score + 1; // Convert 0-4 to 1-5 scale
        totalScore += score + 1;
      });

      const overallScore = Math.round((totalScore / (currentQuiz.questions.length * 5)) * 100);

      return `
        <div class="love-language-result">
          <h3>Overall Relationship Health: ${overallScore}%</h3>
          <p class="description">Your relationship shows ${overallScore >= 80 ? 'excellent' : overallScore >= 60 ? 'good' : 'room for improvement'} health across key areas. Focus on communication, trust, and quality time together to strengthen your bond.</p>
        </div>
      `;
    }

    function showQuizSelection() {
      document.getElementById('quizSelection').style.display = 'block';
      document.getElementById('quizInterface').style.display = 'none';
      document.getElementById('resultsScreen').style.display = 'none';
      
      // Reset quiz state
      currentQuiz = null;
      currentQuestionIndex = 0;
      currentPartner = 'A';
      answers = { A: {}, B: {} };
      selectedAnswer = null;
    }

    function exitQuiz() {
      if (confirm('Are you sure you want to exit this quiz? Your progress will be lost.')) {
        showQuizSelection();
      }
    }

    // Partner tab switching
    document.addEventListener('click', function(e) {
      if (e.target.classList.contains('partner-tab')) {
        const newPartner = e.target.dataset.partner;
        if (newPartner !== currentPartner) {
          // Save current progress and switch
          currentPartner = newPartner;
          document.querySelectorAll('.partner-tab').forEach(tab => {
            tab.classList.remove('active');
          });
          e.target.classList.add('active');
          currentQuestionIndex = 0;
          selectedAnswer = null;
          displayQuestion();
        }
      }
    });
