    // Game Arena JavaScript Functions
    let currentGame = '';
    let gameState = {};
    
    function openGame(gameType) {
      currentGame = gameType;
      var grid = document.querySelector('.intimate-grid');
      if (grid) grid.style.display = 'none';
      var iface = document.getElementById(gameType + '-interface');
      document.body.appendChild(iface);
      iface.style.display = 'block';
      iface.scrollTop = 0;
      initializeGame(gameType);
    }
    
    function closeGame() {
      if (currentGame) {
        document.getElementById(currentGame + '-interface').style.display = 'none';
        currentGame = '';
        gameState = {};
      }
      var grid = document.querySelector('.intimate-grid');
      if (grid) grid.style.display = 'grid';
    }
    
    function initializeGame(gameType) {
      switch(gameType) {
        case 'truth-or-dare':
          initTruthOrDare();
          break;
        case 'would-you-rather':
          initWouldYouRather();
          break;
        case 'story-time':
          initStoryTime();
          break;
        case 'couple-trivia':
          initCoupleTrivia();
          break;
        case 'decision-roulette':
          initDecisionRoulette();
          break;
        case 'compliment-battle':
          initComplimentBattle();
          break;
      }
    }

    function initTruthOrDare() {
      gameState.truthDare = { currentCategory: null, usedPrompts: [] };
    }
    
    function selectTruthDareCategory(category) {
      gameState.truthDare.currentCategory = category;
      gameState.truthDare.usedPrompts = [];
      document.querySelector('.category-selector').style.display = 'none';
      document.getElementById('truth-dare-game').style.display = 'block';
      document.getElementById('truth-dare-prompt').textContent = `Category: ${category.charAt(0).toUpperCase() + category.slice(1)} - Choose Truth or Dare!`;
    }
    
    function nextTruthDare(choice) {
      const category = gameState.truthDare.currentCategory;
      if (!category) return;
      
      if (choice === 'skip' || choice === 'next') {
        document.getElementById('truth-dare-prompt').textContent = 'Choose Truth or Dare for the next round!';
        return;
      }
      
      const prompts = truthDareData[category][choice] || [];
      if (prompts.length === 0) return;
      
      let availablePrompts = prompts.filter((_, index) => !gameState.truthDare.usedPrompts.includes(index));
      if (availablePrompts.length === 0) {
        gameState.truthDare.usedPrompts = [];
        availablePrompts = prompts;
      }
      
      const randomIndex = Math.floor(Math.random() * availablePrompts.length);
      const selectedPrompt = availablePrompts[randomIndex];
      const originalIndex = prompts.indexOf(selectedPrompt);
      gameState.truthDare.usedPrompts.push(originalIndex);
      
      document.getElementById('truth-dare-prompt').textContent = selectedPrompt;
    }

    function initWouldYouRather() {
      gameState.wyr = { currentIndex: 0, usedQuestions: [], playerAChoice: null, playerBChoice: null };
      loadWYRQuestion();
    }
    
    function loadWYRQuestion() {
      let availableQuestions = wouldYouRatherData.filter((_, index) => !gameState.wyr.usedQuestions.includes(index));
      if (availableQuestions.length === 0) {
        gameState.wyr.usedQuestions = [];
        availableQuestions = wouldYouRatherData;
      }
      
      const randomIndex = Math.floor(Math.random() * availableQuestions.length);
      const question = availableQuestions[randomIndex];
      gameState.wyr.currentIndex = wouldYouRatherData.indexOf(question);
      gameState.wyr.usedQuestions.push(gameState.wyr.currentIndex);
      
      document.getElementById('wyr-question').textContent = question.question;
      document.getElementById('wyr-option-a').textContent = question.optionA;
      document.getElementById('wyr-option-b').textContent = question.optionB;
      
      document.getElementById('wyr-result').style.display = 'none';
      document.querySelectorAll('.wyr-option').forEach(btn => {
        btn.style.backgroundColor = '';
        btn.disabled = false;
      });
      gameState.wyr.playerAChoice = null;
      gameState.wyr.playerBChoice = null;
    }
    
    function selectWYROption(option) {
      // This is simplified - in real implementation, you'd have separate selection for each player
      if (!gameState.wyr.playerAChoice) {
        gameState.wyr.playerAChoice = option;

        document.getElementById('wyr-option-' + option).style.backgroundColor = '#4CAF50';

    function startStoryRound() {
      if (gameState.story.timer) clearInterval(gameState.story.timer);
      gameState.story.timeLeft = 60;
      document.getElementById('story-timer').textContent = gameState.story.timeLeft;
      
      gameState.story.timer = setInterval(() => {
        gameState.story.timeLeft--;
        document.getElementById('story-timer').textContent = gameState.story.timeLeft;

          clearInterval(gameState.story.timer);
          alert(`Time's up for Player ${gameState.story.currentPlayer}!`);
        }
      }, 1000);

    function initCoupleTrivia() {
      gameState.trivia = { currentQuestion: 0, score: 0, usedQuestions: [] };
      loadTriviaQuestion();
    }
    
    function loadTriviaQuestion() {
      if (gameState.trivia.currentQuestion >= 15) {
        alert(`Game Over! Final Score: ${gameState.trivia.score}/15`);
        return;
      }
      
      let availableQuestions = triviaQuestions.filter((_, index) => !gameState.trivia.usedQuestions.includes(index));
      if (availableQuestions.length === 0) {
        gameState.trivia.usedQuestions = [];
        availableQuestions = triviaQuestions;
      }
      
      const randomIndex = Math.floor(Math.random() * availableQuestions.length);
      const question = availableQuestions[randomIndex];
      gameState.trivia.currentQuestionData = question;
      gameState.trivia.usedQuestions.push(triviaQuestions.indexOf(question));
      
      document.getElementById('trivia-q').textContent = question.question;
      const options = document.querySelectorAll('.trivia-option');
      question.answers.forEach((answer, index) => {
        if (options[index]) {
          options[index].textContent = answer;
          options[index].style.backgroundColor = '';
          options[index].disabled = false;
        }
      });
    }
    
    function selectTriviaAnswer(answerIndex) {
      const correct = gameState.trivia.currentQuestionData.correct;
      const options = document.querySelectorAll('.trivia-option');
      
      if (answerIndex === correct) {
        gameState.trivia.score++;
        options[answerIndex].style.backgroundColor = '#4CAF50';
        setTimeout(() => alert('Correct!'), 100);
      } else {
        options[answerIndex].style.backgroundColor = '#f44336';
        options[correct].style.backgroundColor = '#4CAF50';
        setTimeout(() => alert('Incorrect! The right answer was highlighted.'), 100);
      }
      
      options.forEach(option => option.disabled = true);
      document.getElementById('trivia-score').textContent = gameState.trivia.score;
    }
    
    function nextTriviaQuestion() {
      gameState.trivia.currentQuestion++;
      setTimeout(() => loadTriviaQuestion(), 500);
    }
    
    // Decision Roulette Game
    function initDecisionRoulette() {
      gameState.roulette = { options: [] };
      document.getElementById('roulette-list').innerHTML = '';
      document.querySelector('.spin-btn').disabled = true;
      document.getElementById('wheel-result').textContent = 'Add options and spin!';
    }
    
    function addRouletteOption() {
      const input = document.getElementById('roulette-input');
      const option = input.value.trim();
      
      if (option && !gameState.roulette.options.includes(option)) {
        gameState.roulette.options.push(option);
        
        const li = document.createElement('li');
        li.textContent = option;
        li.style.cssText = 'color: #b0b8d4; margin: 5px 0; padding: 5px; background: rgba(255,255,255,0.1); border-radius: 5px;';
        document.getElementById('roulette-list').appendChild(li);
        
        input.value = '';
        
        if (gameState.roulette.options.length >= 2) {
          document.querySelector('.spin-btn').disabled = false;
        }
      }
    }
    
    // Add keyboard support for roulette input
    document.addEventListener('DOMContentLoaded', function() {
      // Add enter key support for roulette input
      document.addEventListener('keypress', function(e) {
        if (e.target && e.target.id === 'roulette-input' && e.key === 'Enter') {
          e.preventDefault();
          addRouletteOption();
        }
      });
    });
    
    function spinRoulette() {
      if (gameState.roulette.options.length < 2) return;
      
      document.getElementById('wheel-result').textContent = 'Spinning...';
      
      setTimeout(() => {
        const randomIndex = Math.floor(Math.random() * gameState.roulette.options.length);
        const winner = gameState.roulette.options[randomIndex];
        document.getElementById('wheel-result').textContent = `🎉 ${winner} 🎉`;
      }, 2000);
    }

        roundActive: false
      };
      document.getElementById('p1-compliments').textContent = '0';
      document.getElementById('p2-compliments').textContent = '0';
      document.getElementById('compliment-timer').textContent = '30';
      document.getElementById('compliment-current').textContent = "Player 1's Turn";
    }
    
    function startComplimentRound() {
      if (gameState.compliments.timer) clearInterval(gameState.compliments.timer);
      
      gameState.compliments.timeLeft = 30;
      gameState.compliments.roundActive = true;
      document.getElementById('compliment-timer').textContent = gameState.compliments.timeLeft;
      
      gameState.compliments.timer = setInterval(() => {
        gameState.compliments.timeLeft--;
        document.getElementById('compliment-timer').textContent = gameState.compliments.timeLeft;
        
        if (gameState.compliments.timeLeft <= 0) {
          clearInterval(gameState.compliments.timer);
          gameState.compliments.roundActive = false;
          alert(`Time's up for Player ${gameState.compliments.currentPlayer}!`);
        }
      }, 1000);
    }
    
    function addCompliment() {
      if (!gameState.compliments.roundActive) return;
      
      if (gameState.compliments.currentPlayer === 1) {
        gameState.compliments.p1Score++;
        document.getElementById('p1-compliments').textContent = gameState.compliments.p1Score;
      } else {
        gameState.compliments.p2Score++;
        document.getElementById('p2-compliments').textContent = gameState.compliments.p2Score;
      }
    }
    
    function switchComplimentPlayer() {
      if (gameState.compliments.timer) clearInterval(gameState.compliments.timer);
      
      gameState.compliments.currentPlayer = gameState.compliments.currentPlayer === 1 ? 2 : 1;
      gameState.compliments.timeLeft = 30;
      gameState.compliments.roundActive = false;
      
      document.getElementById('compliment-current').textContent = `Player ${gameState.compliments.currentPlayer}'s Turn`;
      document.getElementById('compliment-timer').textContent = gameState.compliments.timeLeft;
    }
  </script>
